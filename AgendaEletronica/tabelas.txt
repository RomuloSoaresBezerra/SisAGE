CREATE TABLE usuario(
id serial PRIMARY KEY,
nome VARCHAR(100),
nascimento DATE,
sexo CHAR,
email VARCHAR(100) UNIQUE NOT NULL,
senha VARCHAR(100) NOT NULL);

CREATE TABLE agenda(
id serial PRIMARY KEY,
nome VARCHAR(100) NOT NULL,
email VARCHAR(100) NOT NULL,
FOREIGN KEY (email) REFERENCES usuario(email) ON UPDATE CASCADE ON DELETE RESTRICT);

CREATE TABLE compromisso(s
id serial PRIMARY KEY,
data DATE,
hora TIME,
descricao VARCHAR(100),
local VARCHAR(100),
email VARCHAR(100) NOT NULL,
nome VARCHAR(100) NOT NULL,
FOREIGN KEY (email) REFERENCES usuario(email) ON UPDATE CASCADE ON DELETE RESTRICT)
